% Define commands to determine whether we are in a subfile or not
\newcommand{\onlyinsubfile}[1]{
    \ifSubfilesClassLoaded
    { % Only in subfile
        #1
    }
    { % Only in Main File
    }
}
\newcommand{\notinsubfile}[1]{
    \ifSubfilesClassLoaded
    { % Only in subfile  
    }
    { % Only in Main File
        #1
    }
}

% #1: Author #2: Title
\newcommand{\subfilecover}[2]{
    \onlyinsubfile{\author{#1}}
    \onlyinsubfile{\title{#2}}
    \onlyinsubfile{\date{\today}}
}

% #1: Chapter Title
\newcommand{\subfileintro}[1]{
    \onlyinsubfile{\maketitle}
    \onlyinsubfile{\input{"../disclaimer/disclaimer.tex"}}
    \onlyinsubfile{\tableofcontents}
    \notinsubfile{\chapter{#1}}
}
